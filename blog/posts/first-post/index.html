Here’s a quick code snippet to get things started:

import boto3

def list_s3_buckets():
    """List all S3 buckets in the account."""
    s3 = boto3.client('s3')
    response = s3.list_buckets()
    for bucket in response['Buckets']:
        print(f"- {bucket['Name']} ({bucket['CreationDate']})")

list_s3_buckets()
